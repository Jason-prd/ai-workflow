## 🚀 快速开始

### 1. 配置 API Base URL

```javascript
// .env
VITE_API_BASE_URL=http://localhost:8000
```

```javascript
// api/client.js
const API_BASE = import.meta.env.VITE_API_BASE_URL || 'http://localhost:8000';

// 通用请求封装
async function apiRequest(endpoint, options = {}) {
  const token = localStorage.getItem('access_token');
  
  const headers = {
    'Content-Type': 'application/json',
    ...(token && { Authorization: `Bearer ${token}` }),
    ...options.headers,
  };
  
  const response = await fetch(`${API_BASE}${endpoint}`, {
    ...options,
    headers,
  });
  
  if (response.status === 401) {
    // Token 过期，跳转登录
    localStorage.removeItem('access_token');
    window.location.href = '/login';
    throw new Error('Unauthorized');
  }
  
  if (!response.ok) {
    const error = await response.json();
    throw new Error(error.detail || 'Request failed');
  }
  
  return response.json();
}
```

## 📝 登录/注册

### 注册
```javascript
async function register(username, email, password) {
  return apiRequest('/api/auth/register', {
    method: 'POST',
    body: JSON.stringify({ username, email, password }),
  });
}

// 使用
const result = await register('zhangsan', 'zhangsan@example.com', 'password123');
localStorage.setItem('access_token', result.token);
```

## ▶️ 执行工作流

### 手动执行
```javascript
async function executeWorkflow(workflowId, triggerInput = null) {
  return apiRequest(`/api/workflows/${workflowId}/execute`, {
    method: 'POST',
    body: triggerInput ? JSON.stringify(triggerInput) : undefined,
  });
}

// 示例：执行并轮询结果
async function runWorkflowAndWait(workflowId) {
  // 1. 启动执行
  const execution = await executeWorkflow(workflowId, { input: '测试数据' });
  
  // 2. 轮询状态
  while (execution.status === 'running' || execution.status === 'pending') {
    await new Promise(r => setTimeout(r, 1500));
    const status = await getExecutionStatus(execution.id);
    console.log(`进度: ${status.progress_percent}%`, status.current_node_name);
    if (status.status !== 'running' && status.status !== 'pending') break;
  }
  
  // 3. 获取完整结果
  return getExecutionDetail(execution.id);
}
```

## 📁 工作流设计器 - 状态管理建议

### Zustand Store 示例

```javascript
// stores/workflowDesigner.js
import { create } from 'zustand';

export const useWorkflowDesignerStore = create((set, get) => ({
  // 基础信息
  name: '',
  description: '',
  triggerConfig: { type: 'manual' },
  
  // 节点列表
  nodes: [],
  
  // 当前选中的节点
  selectedNodeId: null,
  
  // 操作方法
  setName: (name) => set({ name }),
  setDescription: (description) => set({ description }),
  setTriggerConfig: (triggerConfig) => set({ triggerConfig }),
  
  addNode: (node) => set((state) => ({
    nodes: [...state.nodes, node],
  })),
  
  updateNode: (nodeId, updates) => set((state) => ({
    nodes: state.nodes.map(n => n.id === nodeId ? { ...n, ...updates } : n),
  })),
  
  deleteNode: (nodeId) => set((state) => ({
    nodes: state.nodes.filter(n => n.id !== nodeId),
    selectedNodeId: state.selectedNodeId === nodeId ? null : state.selectedNodeId,
  })),
  
  selectNode: (nodeId) => set({ selectedNodeId: nodeId }),
  
  // 从已有工作流加载
  loadWorkflow: (workflow) => set({
    name: workflow.name,
    description: workflow.description || '',
    triggerConfig: workflow.trigger_config || { type: 'manual' },
    nodes: workflow.nodes || [],
    selectedNodeId: null,
  }),
  
  // 导出为保存格式
  toSaveData: () => {
    const state = get();
    return {
      name: state.name,
      description: state.description || null,
      trigger_config: state.triggerConfig,
      nodes: state.nodes,
    };
  },
}));
```

## ⚠️ 常见问题

### Q: Token 过期了怎么办？
**A:** 前端在收到 401 响应时，应清除本地 Token 并跳转登录页。建议在 API 封装中统一处理。

### Q: 执行超时了怎么办？
**A:** 当前端轮询发现 status 一直为 `running` 超过 60 秒时，可提示用户"执行超时，请稍后刷新状态"。

### Q: 工作流节点最多多少个？
**A:** 当前限制 `MAX_NODES_PER_WORKFLOW = 20`。创建/更新时可通过 `/validate` 接口提前检查。

### Q: 如何获取飞书群聊 ID？
**A:** 飞书机器人接收消息时，事件中会包含 `chat_id`。也可在飞书群设置中查看。

| 环境 | Base URL |
|------|----------|
| 开发 | `http://localhost:8000` |
| 生产 | 根据部署环境配置 |

生产环境请注意：
1. 配置正确的 CORS 策略（后端 `allow_origins`）
2. 使用 HTTPS
3. 生产环境的 `SECRET_KEY` 必须使用随机值
